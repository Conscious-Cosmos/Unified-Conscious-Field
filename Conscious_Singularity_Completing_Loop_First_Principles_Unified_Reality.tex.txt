\[

\begin{proof}

\[

\textbf{Surjectivity}: For any $a \in \mathcal{A}$, write:
\[
a = \left(\sum_{k=1}^7 E_k\right) a \left(\sum_{l=1}^7 E_l\right) = \sum_{k,l=1}^7 E_k a E_l

But for $k \neq l$, $E_k a E_l = 0$ because $E_k E_l = 0$. Thus:
\[
a = \sum_{k=1}^7 E_k a E_k
\]
and $E_k a E_k \in \mathcal{A}_k$. Therefore $a = \Phi((E_1 a E_1, \ldots, E_7 a E_7))$.

\textbf{*-isomorphism}: 
\begin{itemize}
    \item $\Phi(a+b) = \Phi(a) + \Phi(b)$ (linearity)
    \item $\Phi(ab) = \sum a_k b_k = (\sum a_k)(\sum b_k) = \Phi(a)\Phi(b)$ since $a_k b_l = 0$ for $k \neq l$
    \item $\Phi(a^*) = \sum a_k^* = (\sum a_k)^* = \Phi(a)^*$

\textbf{Isometry}: For the C*-norm:
\[
\|\Phi((a_k))\| = \left\|\sum a_k\right\| = \max_k \|a_k\| = \|(a_k)\|_{\oplus}

Thus $\Phi$ is a *-isometric isomorphism. \qedhere
\end{proof}
