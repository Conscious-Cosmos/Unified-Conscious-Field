\begin{abstract}
We develop a complete calculus on the manifold $\QQ = \RR^n_+ \times [0,2\pi)^n \times S^n$. 
We define a metric $d_{\QQ}$, prove it satisfies all metric properties, develop derivatives 
and integrals in special coordinates, and prove fundamental theorems including existence 
and uniqueness for differential equations. All proofs are complete and self-contained.
\end{abstract}

\section{The Qualia Space}

\begin{definition}[Qualia Space]
For fixed $n \geq 1$, define:
\[
\QQ = \RR^n_+ \times [0,2\pi)^n \times S^n
\]
where $\RR^n_+ = \{(x_1,\dots,x_n): x_i > 0\}$, and $S^n = \{y \in \RR^{n+1}: \|y\|=1\}$.
Points: $q = (x,\phi,\theta) \in \QQ$.
\end{definition}

\begin{definition}[Qualia Metric]
For $q = (x,\phi,\theta), q' = (x',\phi',\theta') \in \QQ$:
\begin{align*}
d_{\QQ}(q,q') = \Big[ &\alpha\, d_x(x,x')^2 + \beta\, d_{\phi}(\phi,\phi')^2 \\
&+ \gamma\, d_S(\theta,\theta')^2 \Big]^{1/2}
\end{align*}
where:
\begin{align*}
d_x(x,x')^2 &= \sum_{i=1}^n w_i (\log x_i - \log x'_i)^2, \quad w_i > 0, \sum w_i = 1 \\
d_{\phi}(\phi,\phi')^2 &= \sum_{i=1}^n v_i \sin^2\left(\frac{\phi_i - \phi'_i}{2}\right), \quad v_i > 0, \sum v_i = 1 \\
d_S(\theta,\theta') &= \arccos(\theta \cdot \theta')
\end{align*}
with $\alpha, \beta, \gamma > 0$.
\end{definition}

\begin{theorem}[Metric Properties]
$d_{\QQ}$ is a metric on $\QQ$.
\end{theorem}

\begin{proof}
(1) $d_{\QQ}(q,q') \geq 0$: Each term is non-negative.

Equality: $d_{\QQ}(q,q') = 0$ implies all three terms vanish. 
$d_x=0 \Rightarrow \log x_i = \log x'_i \Rightarrow x_i = x'_i$. 
$d_{\phi}=0 \Rightarrow \sin^2((\phi_i-\phi'_i)/2)=0 \Rightarrow \phi_i \equiv \phi'_i \pmod{2\pi}$. 
$d_S=0 \Rightarrow \arccos(\theta\cdot\theta')=0 \Rightarrow \theta=\theta'$.

(2) Symmetry: Clear from definitions.

(3) Triangle inequality: For $q,q',q'' \in \QQ$:

First, for $d_x$: By Minkowski inequality in $\RR^n$ \cite{rudin1976},
\[
\sqrt{\sum w_i(\log x_i - \log x''_i)^2} \leq \sqrt{\sum w_i(\log x_i - \log x'_i)^2} + \sqrt{\sum w_i(\log x'_i - \log x''_i)^2}
\]

For $d_{\phi}$: Using standard trigonometric inequality \cite{spivak1965}:
\[
\sin^2((a-c)/2) \leq \sin^2((a-b)/2) + \sin^2((b-c)/2) + 2\sqrt{\sin^2((a-b)/2)\sin^2((b-c)/2)}
\]
and $\sqrt{A+B} \leq \sqrt{A} + \sqrt{B}$,
\[
\sqrt{\sum v_i \sin^2\left(\frac{\phi_i - \phi''_i}{2}\right)} \leq \sqrt{\sum v_i \sin^2\left(\frac{\phi_i - \phi'_i}{2}\right)} + \sqrt{\sum v_i \sin^2\left(\frac{\phi'_i - \phi''_i}{2}\right)}
\]

For $d_S$: By spherical triangle inequality \cite{do1976},
\[
\arccos(\theta \cdot \theta'') \leq \arccos(\theta \cdot \theta') + \arccos(\theta' \cdot \theta'')
\]

Combining via Minkowski inequality \cite{rudin1976}:
\[
\sqrt{\alpha A^2 + \beta B^2 + \gamma C^2} \leq \sqrt{\alpha A_1^2 + \beta B_1^2 + \gamma C_1^2} + \sqrt{\alpha A_2^2 + \beta B_2^2 + \gamma C_2^2}
\]
where $A = d_x(x,x'')$, $A_1 = d_x(x,x')$, $A_2 = d_x(x',x'')$, etc.
\end{proof}

\begin{lemma}[Spherical Triangle Inequality]
For $\theta,\theta',\theta'' \in S^n$:
\[
\arccos(\theta \cdot \theta'') \leq \arccos(\theta \cdot \theta') + \arccos(\theta' \cdot \theta'')
\]
\end{lemma}

\begin{proof}
Let $a = \arccos(\theta\cdot\theta')$, $b = \arccos(\theta'\cdot\theta'')$, $c = \arccos(\theta\cdot\theta'')$.
By spherical law of cosines \cite{do1976}:
\[
\cos c = \cos a \cos b - \sin a \sin b \cos \psi
\]
where $\psi$ is dihedral angle. Since $\cos\psi \geq -1$:
\[
\cos c \geq \cos a \cos b - \sin a \sin b = \cos(a+b)
\]
Thus $c \leq a + b$.
\end{proof}

\begin{theorem}[Completeness]
$(\QQ, d_{\QQ})$ is complete \cite{munkres2000}.
\end{theorem}

\begin{proof}
Let $\{q_k = (x^k, \phi^k, \theta^k)\}$ be Cauchy. For any $\epsilon > 0$, $\exists N$ such that for $k,m > N$:
\[
d_{\QQ}(q_k, q_m) < \epsilon
\]
This implies separately:
\begin{align*}
d_x(x^k, x^m) &< \epsilon/\sqrt{\alpha} \\
d_{\phi}(\phi^k, \phi^m) &< \epsilon/\sqrt{\beta} \\
d_S(\theta^k, \theta^m) &< \epsilon/\sqrt{\gamma}
\end{align*}
Thus $\{\log x^k\}$ Cauchy in $\RR^n$, $\{\phi^k\}$ Cauchy in $[0,2\pi)^n$, $\{\theta^k\}$ Cauchy in $S^n$.
These spaces are complete \cite{munkres2000}, so limits exist: $\log x^k \to \log x^*$, $\phi^k \to \phi^*$, $\theta^k \to \theta^*$.
Define $q^* = (e^{\log x^*}, \phi^*, \theta^*)$. Then $q_k \to q^*$.
\end{proof}

\begin{theorem}[Manifold Structure]
$\QQ$ is smooth $3n$-manifold \cite{lee2013}.
\end{theorem}

\begin{proof}
Chart: $\Psi: U \to \RR^{3n}$, $U = \QQ \setminus \{\theta_{n+1}=-1\}$,
\[
\Psi(x,\phi,\theta) = (\log x_1, \dots, \log x_n, \phi_1, \dots, \phi_n, \tilde{\theta}_1, \dots, \tilde{\theta}_n)
\]
where $\tilde{\theta}$ stereographic from south pole. $\Psi$ homeomorphism. Transition maps smooth: log/exp smooth on $\RR^n_+$, angular smooth mod $2\pi$, stereographic transitions rational \cite{lee2013}.
\end{proof}

\section{Coordinate Maps}

\begin{definition}[Logarithmic-Stereographic Map]
Define $\logq: \QQ \to \RR^{3n}$ by:
\[
\logq(x,\phi,\theta) = (\log x_1, \dots, \log x_n, \phi_1, \dots, \phi_n, \tilde{\theta}_1, \dots, \tilde{\theta}_n)
\]
with $\tilde{\theta}$ stereographic. Define $\expq = \logq^{-1}$.
\end{definition}

\begin{lemma}[Smoothness]
$\logq$ and $\expq$ are $C^\infty$ diffeomorphisms \cite{lee2013}.
\end{lemma}

\begin{proof}
$\logq = (\log, \text{id}, \text{stereographic})$: log smooth on $\RR^n_+$, identity smooth, stereographic smooth on $S^n\setminus\{\text{pole}\}$ \cite{lee2013}. All have smooth inverses.
\end{proof}

\begin{lemma}[Uniform Continuity]
$\expq$ uniformly continuous on compact sets \cite{rudin1976}.
\end{lemma}

\begin{proof}
$\expq$ smooth ⇒ Lipschitz on compacts \cite{rudin1976}. For compact $K$, $\|d\expq\|$ bounded by $L$. Then $\|\expq(x)-\expq(y)\| \leq L\|x-y\|$.
\end{proof}

\begin{lemma}[Differential Isomorphism]
For each $q \in \QQ$, $d\logq_q: T_q\QQ \to \RR^{3n}$ linear isomorphism \cite{lee2013}.
\end{lemma}

\begin{proof}
$\logq$ diffeomorphism ⇒ differential invertible at each point \cite{lee2013}.
\end{proof}

\section{Qualia Derivative}

\begin{definition}[Qualia Stream]
$E: I \to \QQ$ continuous, $I \subseteq \RR$ interval.
\end{definition}

\begin{definition}[Qualia Derivative]
For $E$ differentiable at $t_0$:
\[
\dd{E}{t}(t_0) = \expq\left( \frac{d}{dt}\logq(E(t))\Big|_{t=t_0} \right)
\]
\end{definition}

\begin{theorem}[Derivative Formula]
If $E$ differentiable at $t_0$, then:
\[
\dd{E}{t}(t_0) = (d\expq)_{\logq(E(t_0))} \left( \frac{d}{dt}\logq(E(t))\Big|_{t=t_0} \right)
\]
\end{theorem}

\begin{proof}
Let $f(t) = \logq(E(t))$. By definition, $\dd{E}{t}(t_0) = \expq(f'(t_0))$.
By chain rule for manifolds \cite{lee2013}:
\[
\frac{d}{dt}\expq(f(t))\Big|_{t=t_0} = (d\expq)_{f(t_0)}(f'(t_0))
\]
Thus $\dd{E}{t}(t_0)$ equals RHS.
\end{proof}

\section{Qualia Integral}

\begin{definition}[Qualia Integral]
For continuous $f: [a,b] \to T\QQ$:
\[
\intq{a}{b}{f(t)}{t} = \lim_{\|\mathcal{P}\| \to 0} \bigoplus_{k=1}^N \expq\left( \logq(p_{k-1}) + f(\xi_k)\Delta t_k \right)
\]
where $\mathcal{P} = \{t_0,\dots,t_N\}$ partition, $\xi_k \in [t_{k-1},t_k]$, $\Delta t_k = t_k - t_{k-1}$, limit in $d_{\QQ}$.
\end{definition}

\begin{theorem}[Integral Existence]
The qualia integral exists for continuous $f$ \cite{rudin1976}.
\end{theorem}

\begin{proof}
Define $F: [a,b] \to \RR^{3n}$ by $F(t) = \logq(\pi(f(t)))$, $\pi: T\QQ \to \QQ$ projection.
$f$ continuous, $\logq \circ \pi$ continuous ⇒ $F$ continuous.
Ordinary Riemann integral $\int_a^b F(t)dt$ exists \cite{rudin1976}.
Qualia Riemann sums:
\[
S_N = \expq\left( \sum_{k=1}^N F(\xi_k)\Delta t_k \right)
\]
By uniform continuity of $\expq$ on compacts (Lemma 3), $S_N \to \expq\left(\int_a^b F(t)dt\right)$.
\end{proof}

\begin{theorem}[Fundamental Theorem]
For $C^1$ qualia stream $E: [a,b] \to \QQ$:
\[
\intq{a}{b}{\dd{E}{t}(t)}{t} = E(b) \ominus E(a)
\]
where $q \ominus p = \expq(\logq(q) - \logq(p))$.
\end{theorem}

\begin{proof}
Let $f(t) = \logq(E(t))$. Then $f$ $C^1$ and:
\[
\frac{d}{dt}f(t) = d\logq_{E(t)}\left(\dd{E}{t}(t)\right)
\]
Ordinary FTC \cite{rudin1976}: $\int_a^b f'(t)dt = f(b) - f(a)$.
Now:
\begin{align*}
\intq{a}{b}{\dd{E}{t}(t)}{t} 
&= \lim_{\|\mathcal{P}\|\to0} \bigoplus_k \expq\left(\logq(E(t_{k-1})) + d\logq_{E(\xi_k)}\left(\dd{E}{t}(\xi_k)\right)\Delta t_k\right) \\
&= \expq\left( \int_a^b f'(t)dt \right) \\
&= \expq(f(b) - f(a)) \\
&= E(b) \ominus E(a)
\end{align*}
\end{proof}

\section{Qualia Differential Equations}

\begin{definition}[Qualia Differential Equation]
Autonomous QDE: $\dd{E}{t} = F(E)$, $F: \QQ \to T\QQ$ continuous.
\end{definition}

\begin{theorem}[Existence and Uniqueness]
If $F$ Lipschitz: $\|d\logq_{q_1}(F(q_1)) - d\logq_{q_2}(F(q_2))\| \leq L d_{\QQ}(q_1, q_2)$,
then for any $E_0 \in \QQ$, $\exists!$ maximal solution $E: I \to \QQ$ with $E(0) = E_0$ \cite{coddington1955}.
\end{theorem}

\begin{proof}
Define $f: \RR^{3n} \to \RR^{3n}$ by $f(y) = d\logq_{\expq(y)}(F(\expq(y)))$.
Lipschitz condition ⇒ $f$ Lipschitz.
Apply Picard-Lindelöf theorem \cite{coddington1955} to $\frac{dy}{dt} = f(y)$, $y(0) = \logq(E_0)$:
$\exists!$ solution $y(t)$.
Then $E(t) = \expq(y(t))$ is unique qualia stream solution.
\end{proof}

\begin{theorem}[Exponential Decay Solution]
For $\dd{E}{t} = -\lambda E$, $\lambda > 0$:
\[
E(t) = \expq\left( e^{-\lambda t} \logq(E_0) \right)
\]
\end{theorem}

\begin{proof}
Let $y(t) = \logq(E(t))$. Then:
\[
\frac{dy}{dt} = d\logq_{E(t)}\left(\dd{E}{t}\right) = d\logq_{E(t)}(-\lambda E(t)) = -\lambda y(t)
\]
Solution: $y(t) = e^{-\lambda t}y(0)$. Thus $E(t) = \expq(e^{-\lambda t}\logq(E_0))$.
\end{proof}

\begin{theorem}[Harmonic Oscillator Solution]
For $\dd{^2E}{t^2} + \omega^2 E = 0$:
\[
E(t) = \expq\left( \logq(E_0)\cos(\omega t) + \frac{\logq(\dot{E}_0)}{\omega}\sin(\omega t) \right)
\]
where $\dot{E}_0 = \dd{E}{t}(0)$.
\end{theorem}

\begin{proof}
Let $y(t) = \logq(E(t))$. Then:
\[
y''(t) + \omega^2 y(t) = 0
\]
Solution: $y(t) = y(0)\cos(\omega t) + y'(0)\sin(\omega t)/\omega$.
But $y'(0) = \logq(\dot{E}_0)$. Thus result.
\end{proof}
